133 set_door_state (int room_number, int direction, exit_state state)
134 {
135  ROOM_DATA* room = 0;
136  if ((room = vnum_to_room (room_number)))
137  {
138  ROOM_DIRECTION_DATA *door;
139  if ((door = room->dir_option[direction]))
140  {
141  switch (state)
142  {
143  case unlocked_and_open:
144  door->exit_info &= ~(EX_LOCKED | EX_CLOSED);
145  break;
146 
147  case unlocked_and_closed:
148  door->exit_info |= EX_CLOSED;
149  door->exit_info &= ~EX_LOCKED;
150  break;
151 
152  case locked_and_closed:
153  door->exit_info |= (EX_LOCKED | EX_CLOSED);
154  break;
155 
156  case gate_unlocked_and_open:
157  door->exit_info &= ~(EX_LOCKED | EX_CLOSED);
158  door->exit_info |= EX_ISGATE;
159  break;
160 
161  case gate_unlocked_and_closed:
162  door->exit_info |= EX_CLOSED;
163  door->exit_info &= ~EX_LOCKED;
164  door->exit_info |= EX_ISGATE;
165  break;
166 
167  case gate_locked_and_closed:
168  door->exit_info |= (EX_LOCKED | EX_CLOSED);
169  door->exit_info |= EX_ISGATE;
170  break;
171  }
172  return true;
173  }
174  }
175  return false;
176 }
